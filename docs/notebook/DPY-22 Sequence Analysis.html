        position[letter] = [i]

# triplet analysis
triplet = {}
nTriplets = 0
totalTriplets = 0
for i in range(len(dpy)-3):    
    trip = dpy[i: i + 3]
    if trip in triplet.keys():
        triplet[trip] += [i]
    else:
        triplet[trip] = [i]
        nTriplets += 1
    totalTriplets += 1

array = []

for key in sorted(triplet):
    P1 = len(position[key[0]])
    P2 = len(position[key[1]])
    P3 = len(position[key[2]])
    probability = P1*P2*P3/len(dpy)**3
    expected = probability*totalTriplets
    array += [[key, len(triplet[key]), expected]]
triplet_df = pd.DataFrame(array, columns=['Triplet', 'Occurrence', 'Expected'])
triplet_df.sort_values('Occurrence', inplace=True)
